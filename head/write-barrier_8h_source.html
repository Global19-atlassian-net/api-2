153  static V8_INLINE bool TryGetCagedHeap(const void* slot, const void* value,
154  WriteBarrier::Params& params) {
155  params.start = reinterpret_cast<uintptr_t>(value) &
156  ~(api_constants::kCagedHeapReservationAlignment - 1);
157  const uintptr_t slot_offset =
158  reinterpret_cast<uintptr_t>(slot) - params.start;
159  if (slot_offset > api_constants::kCagedHeapReservationSize) {
160  // Check if slot is on stack or value is sentinel or nullptr. This relies
161  // on the fact that kSentinelPointer is encoded as 0x1.
162  return false;
163  }
164  return true;
165  }
166 
167  // Returns whether marking is in progress. If marking is not in progress
168  // sets the start of the cage accordingly.
169  //
170  // TODO(chromium:1056170): Create fast path on API.
171  static bool IsMarking(const HeapHandle&, WriteBarrier::Params&);
172 };

175 struct WriteBarrierTypeForCagedHeapPolicy::ValueModeDispatch<
176  WriteBarrier::ValueMode::kValuePresent> {
177  template <typename HeapHandleCallback>
178  static V8_INLINE WriteBarrier::Type Get(const void* slot, const void* value,
179  WriteBarrier::Params& params,
180  HeapHandleCallback) {
181  bool within_cage = TryGetCagedHeap(slot, value, params);
182  if (!within_cage) {
183  return WriteBarrier::Type::kNone;
184  }
185  if (V8_LIKELY(!params.caged_heap().is_incremental_marking_in_progress)) {
186 #if defined(CPPGC_YOUNG_GENERATION)
187  params.heap = reinterpret_cast<HeapHandle*>(params.start);
188  params.slot_offset = reinterpret_cast<uintptr_t>(slot) - params.start;
189  params.value_offset = reinterpret_cast<uintptr_t>(value) - params.start;
190  return SetAndReturnType<WriteBarrier::Type::kGenerational>(params);
191 #else // !CPPGC_YOUNG_GENERATION
192  return SetAndReturnType<WriteBarrier::Type::kNone>(params);
193 #endif // !CPPGC_YOUNG_GENERATION
194  }
195  params.heap = reinterpret_cast<HeapHandle*>(params.start);
196  return SetAndReturnType<WriteBarrier::Type::kMarking>(params);
197  }

200 template <>
201 struct WriteBarrierTypeForCagedHeapPolicy::ValueModeDispatch<
202  WriteBarrier::ValueMode::kNoValuePresent> {
203  template <typename HeapHandleCallback>
204  static V8_INLINE WriteBarrier::Type Get(const void* slot, const void*,
205  WriteBarrier::Params& params,
206  HeapHandleCallback callback) {
207 #if defined(CPPGC_YOUNG_GENERATION)
208  HeapHandle& handle = callback();
209  if (V8_LIKELY(!IsMarking(handle, params))) {
210  // params.start is populated by IsMarking().
211  params.heap = &handle;
212  params.slot_offset = reinterpret_cast<uintptr_t>(slot) - params.start;
213  // params.value_offset stays 0.
214  if (params.slot_offset > api_constants::kCagedHeapReservationSize) {
215  // Check if slot is on stack.
216  return SetAndReturnType<WriteBarrier::Type::kNone>(params);
217  }
218  return SetAndReturnType<WriteBarrier::Type::kGenerational>(params);
219  }
220 #else // !CPPGC_YOUNG_GENERATION
221  if (V8_LIKELY(!ProcessHeap::IsAnyIncrementalOrConcurrentMarking())) {
222  return SetAndReturnType<WriteBarrier::Type::kNone>(params);
223  }
224  HeapHandle& handle = callback();
225  if (V8_UNLIKELY(!subtle::HeapState::IsMarking(handle))) {
226  return SetAndReturnType<WriteBarrier::Type::kNone>(params);
227  }
228 #endif // !CPPGC_YOUNG_GENERATION
229  params.heap = &handle;
230  return SetAndReturnType<WriteBarrier::Type::kMarking>(params);
231  }

267 struct WriteBarrierTypeForNonCagedHeapPolicy::ValueModeDispatch<
268  WriteBarrier::ValueMode::kValuePresent> {
269  template <typename HeapHandleCallback>
270  static V8_INLINE WriteBarrier::Type Get(const void*, const void* object,
271  WriteBarrier::Params& params,
272  HeapHandleCallback callback) {
273  // The following check covers nullptr as well as sentinel pointer.
274  if (object <= static_cast<void*>(kSentinelPointer)) {
275  return WriteBarrier::Type::kNone;
276  }
277  if (IsMarking(object, &params.heap)) {
278  return SetAndReturnType<WriteBarrier::Type::kMarking>(params);
279  }
